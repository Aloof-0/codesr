# 针对BookInfo增上改查需求的接口定义



# 获取列表数据
# 请求方式：GET
# 请求路径：/books/
# 请求参数：无
# 响应参数：json格式


# 新建单一数据
# 请求方式：POST
# 请求路径：/books/
# 请求参数：json格式或表单请求体
# 响应参数：json格式


# 获取一条数据
# 请求方式：GET
# 请求路径：/books/<pk>/
# 请求参数：路径中的主键值
# 响应参数：json格式


# 更新一条数据
# (全更新) —— 全校验(必要字段必传必校验)
# 请求方式：PUT
# 请求路径：/books/<pk>/
# 请求参数：json格式或表单格式
# 响应参数：json

# (部分更新) —— 部分校验(传什么字段校验什么字段更新什么字段)
# 请求方式：PATCH
# 请求路径：/books/<pk>/
# 请求参数：json格式或表单格式
# 响应参数：json


# 删除一条数据
# 请求方式：DELETE
# 请求路径：/books/<pk>/
# 请求参数：路径中的主键值
# 响应参数：无
